--- layout: post title: Autocomplete category: molecules description: Whenever a field has the ability to autocomplete its value, we should add a hint in form of placeholder and also an icon on the left that shows a spinner while loading from remote --- %h5 Input with autocompletion .form-group#form-group-id %label{ for: 'input-id' } Label: .input-group .input-group-prepend %span.input-group-text#search-icon-id %i.fa.fa-search %i.fas.fa-spinner.fa-spin.d-none %input.form-control#input-id{ type: 'text', placeholder: 'Type to autocomplete...' } :javascript $('#input_id').autocomplete({ appendTo: '#form-group-id', source: 'http://localhost?url_to_remote', search: function(event, ui) { var icon = $('#search-icon-id i:first-child'); icon.addClass('d-none'); icon.next().removeClass('d-none'); }, response: function(event, ui) { var icon = $('#search-icon-id i:first-child'); icon.removeClass('d-none'); icon.next().addClass('d-none'); }, minLength: 2 });
